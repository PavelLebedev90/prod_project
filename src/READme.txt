-инициализация проекта
npm init -y

установить пакет как дев зависимость

-установка webpack webpack-cli определенных версий
npm install -D webpack@5.69.1 webpack-cli@4.9.2

webpack.config.js

-плагин для html
npm install -D html-webpack-plugin@5.5.0

npm install -D typescript@4.5.5 ts-loader@9.2.6
npm install -D ts-node@10.5.0 @types/node@17.0.21 @types/webpack@5.28.0

npm i -D webpack-dev-server@4.7.4 @types/webpack-dev-server@4.7.2
